void setup(){
  izquierda.begin(Parlante_Izquierdo);
  derecha.begin(Parlante_Derecho);
  pinMode(Led_Encendido, OUTPUT);
  pinMode(Led_Izquierdo, OUTPUT);
  pinMode(Led_Derecho, OUTPUT);
}

void suenanDos(int nota1, int nota2, int tiempo){
  int retardo = tiempo - 4;
  izquierda.play(nota1, tiempo);
  derecha.play(nota2, tiempo);
  digitalWrite(Led_Izquierdo, HIGH);
  digitalWrite(Led_Derecho, HIGH);
  delay(retardo);
  digitalWrite(Led_Izquierdo, LOW);
  digitalWrite(Led_Derecho, LOW);
  delay(5);
}

void suenaUno(int nota1, int tiempo){
  int retardo = tiempo - 4;
  izquierda.play(nota1, tiempo);
  digitalWrite(Led_Izquierdo, HIGH);
  delay(retardo);
  digitalWrite(Led_Izquierdo, LOW);
  delay(5);
}

void silencio(int tiempo){
  int retardo = tiempo - 4;
  delay(retardo);
}
